% Split journal papers by review status using the BibTeX field note={under review}
\defbibcheck{journalpublished}{%
	\ifentrytype{article}{%
		\iffieldundef{note}{}{%
			\iffieldequalstr{note}{under review}{\skipentry}{}%
		}%
	}{\skipentry}%
}

\defbibcheck{journalunderreview}{%
	\ifentrytype{article}{%
		\iffieldequalstr{note}{under review}{}{\skipentry}%
	}{\skipentry}%
}

\defbibcheck{conferencepublished}{%
	\ifentrytype{inproceedings}{%
		\iffieldundef{note}{}{%
			\iffieldequalstr{note}{under review}{\skipentry}{}%
		}%
	}{\skipentry}%
}

\defbibcheck{conferenceunderreview}{%
	\ifentrytype{inproceedings}{%
		\iffieldequalstr{note}{under review}{}{\skipentry}%
	}{\skipentry}%
}

\printbibliography[heading={subbibliography},title={Peer-Reviewed Journal Papers},check=journalpublished]
\printbibliography[heading={subbibliography},title={Peer-Reviewed Journal Papers \textit{Under Review}},check=journalunderreview]
